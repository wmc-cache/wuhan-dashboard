### 1. 工会会员数

**数据结构**:
```typescript
interface MemberCountItem {
  area: string;          // 行政区名称
  orgCount: number;      // 分工会总数（个）
  memberCount: number;   // 会员总数（万人）
}

interface MemberCountResponse {
  data: MemberCountItem[];
  leftMax?: number;      // 左侧坐标上限（分工会总数）
  rightMax?: number;     // 右侧坐标上限（会员总数）
}
```

### 2.年度工会创建数

**数据结构**:
```typescript
interface MonthlyData {
  total: number[];    // 12个月的工会总数
  added: number[];    // 12个月的新增工会数
}

interface YearCreateResponse {
  years: number[];                           // 可选年份列表
  dataByYear: Record<string, MonthlyData>;   // 按年份分组的数据
  yMax?: number;                             // Y轴最大值
  initialYear?: number;                      // 默认显示年份
}
```
